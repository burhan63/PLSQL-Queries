create or replace function total_emp
return number
is 
begin 
  select count(employee_id) from 
  employees;
  return total_emp;
end;

=== 2nd function ==
CREATE OR REPLACE FUNCTION total_emp  
RETURN number IS  
   total number(10) := 0;  
BEGIN  
   SELECT count(*) into total  
   FROM employees;  
    RETURN total;  
END;
