-------- Explicit Cursor -----------
DECLARE
  product_id  products_orders.p_id%type;
  product_no  products_orders.p_no%type;
  product_name products_orders.p_name%type;
  product_brand products_orders.p_brand%type;
  product_headquarter products_orders.p_headquarter%type;  
  
  CURSOR product_description 
  IS
  select po.p_id,po.p_no,
  po.p_name,po.p_brand,
  po.p_headquarter from 
  products_orders po;
  
BEGIN 
  for rec in(select po.p_id,po.p_no,
  po.p_name,po.p_brand,
  po.p_headquarter from 
  products_orders po)loop
  
  dbms_output.put_line(rec.p_id || ' ' || rec.p_no || ' ' || rec.p_name || ' ' ||  rec.p_brand || ' ' ||
  ' ' || rec.p_headquarter);
  end loop;
  
END;
