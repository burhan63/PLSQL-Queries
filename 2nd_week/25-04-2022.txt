-- online terminal --
create table customer(cust_id number primary key,
cust_fname varchar2(50),
cust_lname varchar2(50),
cust_gender char(1),
cust_contact number,
cust_city varchar2(30),
cust_nationality varchar2(40),
cust_address varchar2(100)
);

select * from customer;

select * from customer order by cust_id asc;

select cust_fname, cust_lname from customer where cust_contact = 3112329374;


-- new table that named as product_order --
select * from product_order;

select * from customer
inner join product_order on
customer.cust_id = product_order.order_id;

-- sub query -- 
select * from product_order where order_price < 
(select avg(order_price) as average_price from product_order where order_id = 30);

select avg(order_price) as average_price from product_order where order_price in 
(select order_price from product_order);

-- sub query using create table -- 
create table product_order_done as select * from product_order;
select * from product_order_done;
select count(*) from product_order_done;
select count(order_scanNo) from product_order_done;
select min(order_price) from product_order_done;
select max(order_price) from product_order_done;
select stddev(order_price) from product_order_done;
select variance(order_price) from product_order_done;


-- subquery by using insert table --

update product_order set order_price = order_price + (select avg(order_price) from product_order);
